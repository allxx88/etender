{{/* Функция для получения description */}} {{- $ctx := . -}} {{- $description := "" -}} {{- if $ctx.Description -}} {{- $description = $ctx.Description -}} {{- else if $ctx.Content -}} {{- $description = $ctx.Content | plainify | truncate 160 -}} {{- else if $ctx.Site.Params.siteDescription -}} {{- $description = $ctx.Site.Params.siteDescription -}} {{- end -}} {{- return $description -}}
